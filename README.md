SELECT me.name as My Name, me.email as Email, STRING_AGG(edu.degree, ',') as Degrees, STRING_AGG(hob.name, ',') as Hobbies
FROM my_info me
JOIN education edu on edu.person_id = me.id
JOIN hobbies hob on hob.person_id = me.id
